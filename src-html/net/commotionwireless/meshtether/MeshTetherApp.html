086            /**
087             * Initializes PreferenceManager, NotificationManager, WifiManager
088             * and {@link net.commotionwireless.meshtether.NativeHelper}.
089             * 
090             * @see android.app.Application#onCreate()
091             */
092            @Override
093            public void onCreate() {
094                    super.onCreate();
095                    Log.d(TAG, "onCreate");
096                    NativeHelper.setup(this);
097    
098                    // initialize default values if not done this in the past
099                    PreferenceManager.setDefaultValues(this, R.xml.preferences, false);
100                    prefs = PreferenceManager.getDefaultSharedPreferences(this);
101    
102                    // if IP address isn't set, generate one
103                    if (prefs.getString(getString(R.string.adhoc_ip), "").equals("")) {
104                            SharedPreferences.Editor e = prefs.edit();
105                            String myIP = "10." + String.valueOf((int)(Math.random() * 254))
106                            + "." + String.valueOf((int)(Math.random() * 254))
107                            + "." + String.valueOf((int)(Math.random() * 254));
108                            e.putString(getString(R.string.adhoc_ip), myIP);
109                            e.commit();
110                            Log.i(TAG, "Generated IP: " + myIP);
111                    }
112    
113                    notificationManager = (NotificationManager) getSystemService(Context.NOTIFICATION_SERVICE);
114                    toast = Toast.makeText(this, "", Toast.LENGTH_SHORT);
115                    notification = new Notification(R.drawable.comlogo_sm_on, getString(R.string.notify_running), 0);
116                    notification.flags |= Notification.FLAG_ONGOING_EVENT;
117                    notificationClientAdded = new Notification(android.R.drawable.stat_sys_warning,
118                                    getString(R.string.notify_client), 0);
119                    notificationClientAdded.flags = Notification.FLAG_AUTO_CANCEL;
120                    notificationError = new Notification(R.drawable.barnacle_error,
121                                    getString(R.string.notify_error), 0);
122                    notificationError.setLatestEventInfo(this,
123                                    getString(R.string.app_name),
124                                    getString(R.string.notify_error),
125                                    PendingIntent.getActivity(this, 0, new Intent(this, StatusActivity.class), 0));
126                    notificationError.flags = Notification.FLAG_AUTO_CANCEL;
127    
128                    activeProfile = getString(R.string.defaultprofile);
129    
130                    mJsonInfo = new JsonInfo();
131                    profiles = new CopyOnWriteArraySet<String>();
132                    profiles.add(activeProfile);
133                    profileProperties = new HashMap<String, String>();
134    
135                    wifiManager = (WifiManager) getSystemService(Context.WIFI_SERVICE);
136            }

226            static void broadcastState(Context ctx, int state) {
227                    Intent intent = new Intent(ACTION_CHANGED);
228                    intent.putExtra("state", state);
229                    ctx.sendBroadcast(intent, "net.commotionwireless.meshtether.ACCESS_STATE");
230            }

245            void clientAdded(MeshService.ClientData cd) {
246                    if (prefs.getBoolean("client_notify", false)) {
247                            notificationClientAdded.defaults = 0;
248                            if (prefs.getBoolean("client_light", false)) {
249                                    notificationClientAdded.flags |= Notification.FLAG_SHOW_LIGHTS;
250                                    notificationClientAdded.ledARGB = 0xffffff00; // yellow
251                                    notificationClientAdded.ledOnMS = 500;
252                                    notificationClientAdded.ledOffMS = 1000;
253                            } else {
254                                    notificationClientAdded.flags &= ~Notification.FLAG_SHOW_LIGHTS;
255                            }
256                            String sound = prefs.getString("client_sound", null);
257                            if (sound == null) {
258                                    notificationClientAdded.defaults |= Notification.DEFAULT_SOUND;
259                                    // | Notification.DEFAULT_VIBRATE // requires permission
260                            } else {
261                                    if (sound.length() > 0)
262                                            notificationClientAdded.sound = Uri.parse(sound);
263                            }
264                            Intent notificationIntent = new Intent(this, StatusActivity.class);
265                            notificationIntent.setAction(ACTION_CLIENTS);
266                            PendingIntent contentIntent = PendingIntent.getActivity(this, 0, notificationIntent, 0);
267                            notificationClientAdded.setLatestEventInfo(this, getString(R.string.app_name),
268                                            getString(R.string.notify_client) + " " + cd.toNiceString(), contentIntent);
269                            notificationManager.notify(NOTIFY_CLIENT, notificationClientAdded);
270                    }
271    
272                    if (linksActivity != null)
273                            linksActivity.update();
274            }

280            void processStarted() {
281                    Log.w(TAG, "processStarted");
282                    Intent notificationIntent = new Intent(this, StatusActivity.class);
283                    PendingIntent contentIntent = PendingIntent.getActivity(this, 0, notificationIntent, 0);
284                    notification.setLatestEventInfo(this, getString(R.string.app_name),
285                                    getString(R.string.notify_running), contentIntent);
286                    notificationManager.notify(NOTIFY_RUNNING, notification);
287                    service.startForegroundCompat(NOTIFY_RUNNING, notification);
288            }
289    
290            void processStopped() {
291                    Log.w(TAG, "processStopped");
292                    notificationManager.cancel(NOTIFY_RUNNING);
293                    notificationManager.cancel(NOTIFY_CLIENT);
294                    if (service != null) service.stopSelf();
295                    service = null;
296                    updateStatus();
297            }

299            void failed(int err) {
300                    if (statusActivity != null) {
301                            if (err == ERROR_ROOT) {
302                                    statusActivity.showDialog(StatusActivity.DLG_ROOT);
303                            } else if (err == ERROR_SUPPLICANT) {
304                                    statusActivity.showDialog(StatusActivity.DLG_SUPPLICANT);
305                            } else if (err == ERROR_OTHER) {
306                                    statusActivity.getTabHost().setCurrentTab(0); // show links
307                                    statusActivity.showDialog(StatusActivity.DLG_ERROR);
308                            }
309                    }
310                    if ((statusActivity == null) || !statusActivity.hasWindowFocus()) {
311                            Log.d(TAG, "notifying error");
312                            notificationManager.notify(NOTIFY_ERROR, notificationError);
313                    }
314            }

316            /**
317             * find default route interface
318             * @return true iff an active WAN interface was found.
319             */
320            protected boolean findIfWan() {
321                    // TODO move to Util.java and actually detect if there is a GSM/CDMA net connection
322                    String if_wan = prefs.getString(getString(R.string.if_wan), "");
323                    if (if_wan.length() != 0) return true;
324    
325                    // must find mobile data interface
326                    ArrayList<String> routes = Util.readLinesFromFile("/proc/net/route");
327                    for (int i = 1; i < routes.size(); ++i) {
328                            String line = routes.get(i);
329                            String[] tokens = line.split("\\s+");
330                            if (tokens[1].equals("00000000")) {
331                                    // this is our default route
332                                    if_wan = tokens[0];
333                                    break;
334                            }
335                    }
336                    if (if_wan.length() != 0) {
337                            updateToast(getString(R.string.wanok) + if_wan, false);
338                            prefs.edit().putString(getString(R.string.if_wan), if_wan).commit();
339                            return true;
340                    }
341                    // it might be okay in local mode
342                    return prefs.getBoolean("wan_nowait", false);
343            }

345            /**
346             * Stores a LAN interface
347             * @param found_if_lan the active LAN interface reference
348             */
349            protected void foundIfLan(String found_if_lan) {
350                    String if_lan = prefs.getString(getString(R.string.if_lan), "");
351                    if (if_lan.length() == 0) {
352                            updateToast(getString(R.string.lanok) + found_if_lan, false);
353                    }
354                    // NOTE: always use the name found by the process
355                    if_lan = found_if_lan;
356                    prefs.edit().putString(getString(R.string.if_lan), if_lan).commit();
357            }

368            void showProgressMessage(String messageText) {
369                    Log.i(TAG, "MSG_PROGRESSDIALOG");
370                    if (messageText == null) messageText = "(null)";
371                    // TODO remove these null check and fix the actual bug! these should
372                    // always exist when this is called
373                    if (statusActivity == null) {
374                            Log.e(TAG, "statusActivity is null!");
375                            return;
376                    }
377                    if (statusActivity.mProgressDialog == null) {
378                            Log.e(TAG, "statusActivity.mProgressDialog is null!");
379                            return;
380                    }
381                    statusActivity.mProgressDialog.setMessage(messageText);
382                    if ( !statusActivity.mProgressDialog.isShowing())
383                            statusActivity.mProgressDialog.show();
384            }
385

387                    Log.i(TAG, "MSG_PROGRESSDIALOG_DISMISS");
