208            @Override
209            public void onDestroy() {
210                    if (state != STATE_STOPPED)
211                            Log.e(TAG, "service destroyed while running!");
212                    // ensure we clean up
213                    stopProcess();
214                    state = STATE_STOPPED;
215                    app.processStopped();
216                    wakeLock.release();
217    
218                    mShell.stopShell();
219                    
220                    try {
221                            unregisterReceiver(connectivityReceiver);
222                    } catch (Exception e) {
223                            // ignore
224                    }
225    
226                    singleton = null;
227                    super.onDestroy();
228            }

253            private void handle(Message msg) {
254                    switch (msg.what) {
255                    case MSG_SHOW_PROGRESSDIALOG:
256                            Log.i(TAG, "handle MSG_SHOW_PROGRESSDIALOG");
257                            app.showProgressMessage((String)msg.obj);
258                            break;
259                    case MSG_HIDE_PROGRESSDIALOG:
260                            Log.i(TAG, "handle MSG_HIDE_PROGRESSDIALOG");
261                            app.hideProgressDialog();
262                            break;
263                    case MSG_EXCEPTION:
264                            Log.i(TAG, "handle MSG_EXCEPTION");
265                            if (state == STATE_STOPPED) return;
266                            Throwable thr = (Throwable)msg.obj;
267                            log(true, getString(R.string.exception) + " " + thr.getMessage());
268                            Log.e(TAG, "Exception " + thr.getMessage() + " " + Log.getStackTraceString(thr));
269                            stopProcess();
270                            mShell.stopShell();
271                            state = STATE_STOPPED;
272                            break;
273                    case MSG_ERROR:
274                            Log.i(TAG, "handle MSG_ERROR");
275                            if (state == STATE_STOPPED) return;
276                            if (WifiShellProcess == null) return; // don't kill it again...
277                            if (msg.obj != null) {
278                                    String line = (String)msg.obj;
279                                    log(true, line); // just dump it and ignore it
280                            } else {
281                                    // no message, means process died
282                                    log(true, getString(R.string.unexpected));
283                                    stopProcess();
284                                    mShell.stopShell();
285    
286                                    if ((state == STATE_STARTING)) {
287                                            String err = log.toString();
288                                            if (isRootError(err)) {
289                                                    app.failed(MeshTetherApp.ERROR_ROOT);
290                                            } else if (isSupplicantError(err)) {
291                                                    app.failed(MeshTetherApp.ERROR_SUPPLICANT);
292                                            } else {
293                                                    app.failed(MeshTetherApp.ERROR_OTHER);
294                                            }
295                                    } else {
296                                            app.failed(MeshTetherApp.ERROR_OTHER);
297                                    }
298                                    state = STATE_STOPPED;
299                            }
300                            break;
301                    case MSG_OUTPUT:
302                            Log.i(TAG, "handle MSG_OUTPUT");
303                            if (state == STATE_STOPPED) return;
304                            if (WifiShellProcess == null) return; // cut the gibberish
305                            String line = (String)msg.obj;
306                            if (line == null) {
307                                    // ignore it, wait for MSG_ERROR(null)
308                                    break;
309                            }
310                            if (line.startsWith("WIFI: OK")) {
311                                    // WIFI: OK <IFNAME> <MAC>
312                                    String[] parts = line.split(" +");
313                                    if_lan = parts[2];
314                                    if_mac = Util.MACAddress.parse(parts[3]);
315                                    if (state == STATE_STARTING) {
316                                            state = STATE_RUNNING;
317                                            log(false, getString(R.string.running));
318                                            clients.clear();
319                                            stats.init(Util.fetchTrafficData(if_lan));
320                                            //app.foundIfLan(if_lan); // this will allow 3G <-> 4G with simple restart
321                                            app.processStarted();
322                                            mHandler.sendEmptyMessage(MSG_ASSOC);
323                                            mHandler.sendEmptyMessage(MSG_HIDE_PROGRESSDIALOG);
324                                    }
325                            } else {
326                                    log(false, line);
327                            }
328                            break;
329                    case MSG_START:
330                            Log.i(TAG, "handle MSG_START");
331    
332                            if (state != STATE_STOPPED) return;
333                            log.clear();
334    
335                            if ( !setMeshProfile())
336                                    break;
337                            log(false, getString(R.string.starting) + " " + activeSSID);
338    
339                            // TODO make this only overwrite on upgrade to new version
340                            log(false, getString(R.string.unpacking));
341                            if (!NativeHelper.unzipAssets(this) || !NativeHelper.installBusyboxSymlinks()) {
342                                    log(true, getString(R.string.unpackerr));
343                                    state = STATE_STOPPED;
344                                    break;
345                            }
346    
347                            if ( !wifiManager.isWifiEnabled()) {
348                                    log(false, getString(R.string.enablingwifi));
349                                    wifiManager.setWifiEnabled(true); // this will send MSG_NETSCHANGE
350                            }
351                            if (wifiManager.getConnectionInfo().getIpAddress() != 0) // if connected, disconnect
352                                    wifiManager.disconnect(); // this will send MSG_NETSCHANGE
353    
354                            mShell.setForkCommand(NativeHelper.SU_C_FORK, buildEnvFromPrefs());
355                            mShell.startShell();
356                            Shell.waitForShellToStart(mShell);
357                            
358                            state = STATE_STARTING;
359                            // FALL THROUGH!
360                    case MSG_NETSCHANGE:
361                            Log.i(TAG, "handle MSG_NETSCHANGE");
362                            int wifiState = wifiManager.getWifiState();
363                            String wifiStateString = getWifiStateString(wifiState);
364                            if (wifiState == WifiManager.WIFI_STATE_ENABLED) {
365                                    // wifi is good (or lost), we can start now...
366                                    if ((state == STATE_STARTING) && (WifiShellProcess == null)) {
367                                            if (app.findIfWan()) {
368                                                    // TODO if WAN found with checkUplink(), then setup Hna4 routing
369                                                    log(false, "Found active WAN interface");
370                                            } else {
371                                                    ///log(true, getString(R.string.wanerr));
372                                                    //state = STATE_STOPPED;
373                                                    //break;
374                                                    log(false, "No active WAN interface found");
375                                            }
376    
377                                            log(false, "Enabling wifi for " + activeSSID);
378                                            WifiConfiguration wc = new WifiConfiguration();
379                                            wc.priority = 100000;
380                                            wc.hiddenSSID = true; // Android won't see Adhoc SSIDs
381                                            wc.status = WifiConfiguration.Status.ENABLED;
382                                            wc.allowedKeyManagement.set(WifiConfiguration.KeyMgmt.NONE);
383                                            wc.SSID = activeSSID;
384                                            wc.BSSID = activeBSSID;
385                                            currentNetId = wifiManager.addNetwork(wc);
386                                            if(currentNetId < 0) {
387                                                    System.out.println("Unable to add network configuration for " + activeSSID);
388                                                    return;
389                                            }
390                                            wifiManager.enableNetwork(currentNetId, true);
391    
392                                            // set up the static IP settings
393                                            // TODO read and store the current settings so they can be restored on stop
394                                            log(false, "Setting up IP config");
395                                            final ContentResolver cr = getContentResolver();
396                                            Settings.System.putInt(cr, Settings.System.WIFI_USE_STATIC_IP, 1);
397                                            Settings.System.putString(cr, Settings.System.WIFI_STATIC_IP, activeIP);
398                                            Settings.System.putString(cr, Settings.System.WIFI_STATIC_NETMASK, activeNetmask);
399                                            Settings.System.putString(cr, Settings.System.WIFI_STATIC_GATEWAY, activeIP);
400                                            Settings.System.putString(cr, Settings.System.WIFI_STATIC_DNS1, activeDNS);
401    
402                                            // meshing works very poorly with wifi sleep enabled
403                                            // TODO check if this sticks after the app quits, if so
404                                            // save the current value and restore it on stopping
405                                            Settings.System.putInt(cr, Settings.System.WIFI_SLEEP_POLICY,
406                                                            Settings.System.WIFI_SLEEP_POLICY_NEVER);
407    
408                                            if (!startProcess()) {
409                                                    log(true, getString(R.string.starterr));
410                                                    state = STATE_STOPPED;
411                                                    break;
412                                            }
413                                    } // if not checkUpLink then we simply wait...
414                            } else { // not WIFI_STATE_ENABLED
415                                    app.updateToast("WIFI: " + wifiStateString, false); log(false, "WIFI: " + wifiStateString);
416                                    switch (state) {
417                                    case STATE_RUNNING:
418                                            app.updateToast("STATE_RUNNING", true); log(false, "STATE_RUNNING");
419                                            stopProcess(); // this tears down everything
420                                            mShell.stopShell();
421                                            state = STATE_STARTING;
422                                            break;
423                                    case STATE_STARTING:
424                                            app.updateToast("STATE_STARTING", false); log(false, "STATE_STARTING");
425                                            break;
426                                    case STATE_STOPPED:
427                                            app.updateToast("STATE_STOPPED", false); log(false, "STATE_STOPPED");
428                                            break;
429                                    default:
430                                            app.updateToast("Something went wrong...", false);      log(false, "Something went wrong...");
431                                    }
432                            }
433                            break;
434                    case MSG_STOP:
435                            Log.i(TAG, "handle MSG_STOP");
436                            if (state == STATE_STOPPED) return;
437                            stopProcess();
438                            mShell.stopShell();
439                            wifiManager.disableNetwork(currentNetId);
440                            wifiManager.removeNetwork(currentNetId);
441                            final ContentResolver cr = getContentResolver();
442                            Settings.System.putInt(cr, Settings.System.WIFI_USE_STATIC_IP, 0);
443                            log(false, getString(R.string.stopped));
444                            state = STATE_STOPPED;
445                            break;
446                    case MSG_ASSOC:
447                            Log.i(TAG, "handle MSG_ASSOC");
448                            if (state != STATE_RUNNING) return;
449                            if (tellProcess("WLAN")) {
450                                    app.updateToast(getString(R.string.beaconing), true);
451                            }
452                            if (clients.isEmpty() && app.prefs.getBoolean("lan_autoassoc", false)) {
453                                    mHandler.removeMessages(MSG_ASSOC);
454                                    // rebeacon, in 5 seconds
455                                    mHandler.sendEmptyMessageDelayed(MSG_ASSOC, 5000);
456                            }
457                            break;
458                    case MSG_STATS:
459                            //Log.i(TAG, "handle MSG_STATS");
460                            mHandler.removeMessages(MSG_STATS);
461                            if (state != STATE_RUNNING || if_lan.length() == 0) return;
462                            stats.update(Util.fetchTrafficData(if_lan));
463                            break;
464                    }
465                    app.updateStatus();
466                    if (state == STATE_STOPPED) {
467                            app.processStopped();
468                            app.hideProgressDialog();
469                    }
470            }

472            protected void log(boolean error, String msg) {
473                    android.text.format.Time time = new android.text.format.Time();
474                    time.setToNow();
475                    Log.i(TAG, "log: " + msg);
476                    log.append(COLOR_TIME, time.format("%H:%M:%S\t"))
477                    .append(error ? COLOR_ERROR : COLOR_LOG, msg)
478                    .append("\n");
479                    Message message = mHandler.obtainMessage();
480                    message.what = MSG_SHOW_PROGRESSDIALOG;
481                    message.obj = msg;
482                    mHandler.sendMessage(message);
483            }

485            /** Worker Threads */
486            private class OutputMonitor implements Runnable {
487                    private final java.io.BufferedReader br;
488                    private final int msg;
489                    public OutputMonitor(int t, java.io.InputStream is) {
490                            br = Util.toReader(is);
491                            msg = t;
492                    }
493                    @Override
494                    public void run() {
495                            try{
496                                    String line;
497                                    do {
498                                            line = br.readLine();
499                                            mHandler.obtainMessage(msg, line).sendToTarget(); // NOTE: the last null is also sent!
500                                    } while(line != null);
501                            } catch (Exception e) {
502                                    mHandler.obtainMessage(MSG_EXCEPTION, e).sendToTarget();
503                            }
504                    }
505            }

531            /** Prepare env vars for ./wifi from preferences */
532            protected String[] buildEnvFromPrefs() {
533                    ArrayList<String> envlist = new ArrayList<String>();
534    
535                    // get the existing environment first, since many programs like 'su' require 
536                    // env vars like LD_LIBRARY_PATH to be set
537                    Map<String, String> env = System.getenv();
538                    for (String envName : env.keySet()) {
539                            if (envName.equals("LD_LIBRARY_PATH")) {
540                                    // set LD_LIBRARY_PATH to load olsrd plugins
541                                    envlist.add(envName + "=" + NativeHelper.app_bin.getAbsolutePath() + ":" + env.get(envName));                           
542                            } else {
543                                    envlist.add(envName + "=" + env.get(envName));
544                            }
545                    }
546    
547                    // initialize default values if not done this in the past
548                    PreferenceManager.setDefaultValues(this, R.xml.preferences, false);
549                    SharedPreferences prefs = PreferenceManager.getDefaultSharedPreferences(this);
550    
551                    final int[] ids = SettingsActivity.prefids;
552                    for (int i = 0; i < ids.length; ++i) {
553                            String k = getString(ids[i]);
554                            String v = prefs.getString(k, null);
555                            if (v != null && v.length() != 0) {
556                                    // TODO some chars need to be escaped, but this seems to add "" to the ESSID name
557                                    //                      if (ids[i] == R.string.lan_essid) {
558                                    //                              v = '"'+v+'"';
559                                    //                      }
560                                    // Special handling for Mesh Prefix 
561                                    if (k.equalsIgnoreCase(getString(R.string.adhoc_ip)))
562                                    {
563                                            if (v.matches("^[0-9]{1,3}$"))
564                                            {
565                                                    // Convert the Mesh Prefix into a Mesh IP.
566                                                    WifiManager wifiManager = (WifiManager)getSystemService(Context.WIFI_SERVICE);
567                                                    if (wifiManager != null)
568                                                    {
569                                                            WifiInfo wifiInfo = wifiManager.getConnectionInfo();
570                                                            if (wifiInfo != null)
571                                                            {
572                                                                    Log.d(TAG, "Mac addr: " + wifiInfo.getMacAddress());
573                                                                    MACAddress macAddress = MACAddress.parse(wifiInfo.getMacAddress());
574                                                                    v += "." + macAddress.getOctet(4) + "." + macAddress.getOctet(5) + "." + macAddress.getOctet(6);
575                                                            }
576                                                    }
577                                            }
578                                    }
579                                    envlist.add("brncl_" + k + "=" + v);
580                            }
581                    }
582                    // not included in prefids are checkboxes
583                    final int[] checks = SettingsActivity.checks;
584                    for (int i = 0; i < checks.length; ++i) {
585                            String k = getString(checks[i]);
586                            if (prefs.getBoolean(k, false))
587                                    envlist.add("brncl_" + k + "=1");
588                    }
589                    envlist.add("brncl_path=" + NativeHelper.app_bin.getAbsolutePath());
590                    envlist.add("olsrd_conf_path=" + activeOlsrdConf);
591                    
592                    String[] ret = envlist.toArray(new String[0]);
593                    for (String s : ret) {
594                            Log.i(TAG, "env var: " + s);
595                    }
596                    return ret;
597            }

605            private String generateIP(String base, String netmask) {
606                    String[] netmaskBytes = netmask.split("[.]+");
607                    String[] baseBytes = base.split("[.]+");
608                    // start at byte 1 since a completely random IP would be stupid
609                    String ret = baseBytes[0];
610                    for (int i=1; i<baseBytes.length; i++) {
611                            // if the netmask byte doesn't exist or is zero, generate that byte
612                            ret += ".";
613                            if (baseBytes[i].equals("0") && (i >= netmaskBytes.length || netmaskBytes[i].equals("0")))
614                                    ret += String.valueOf((int)(Math.random() * 254));
615                            else
616                                    ret += Byte.parseByte(baseBytes[i]);
617                    }
618                    return ret;
619            }

699            private void stopProcess() {
700                    /*
701                     * TODO: UN copy and paste!!
702                     */
703                    if (state != STATE_STOPPED) {
704                            app.showProgressMessage(R.string.servicestopping);
705                            ShellProcess StopOlsrProcess = null;
706                            
707                            StopOlsrProcess = new ShellProcess("MSG_STOP_OLSRD_OUTPUT", NativeHelper.STOP_OLSRD, mShell);
708                            StopOlsrProcess.setHandler(mHandler, MSG_STOP_OLSRD_OUTPUT);
709                            if (!StopOlsrProcess.runSynchronous()) {
710                                    Log.w(TAG, "Could not StopOlsrProcess");
711                            }
712                            if (WifiShellProcess != null) {
713                                    // first, just close the stream
714                                    if (state != STATE_STOPPED) {
715                                            try {
716                                                    WifiShellProcess.stop();
717                                            } catch (Exception e) {
718                                                    Log.w(TAG, "Exception while closing WifiShellProcess: " + e.toString());
719                                            }
720                                    }
721                                    WifiShellProcess = null;
722                            }
723                            log(false, "Disconnecting from " + activeSSID);
724                            wifiManager.disconnect();
725                            log(false, "Releasing wifi lock");
726                            wifiLock.release();
727                            clients.clear();
728                            app.linksActivity.update();
729                            app.hideProgressDialog();
730                    }
731            }
