001    package net.commotionwireless.meshtether;
002    
003    import android.content.BroadcastReceiver;
004    import android.content.Context;
005    import android.content.Intent;
006    import android.util.Log;
007    
008    public class ToggleReceiver extends BroadcastReceiver {
009            final static String TAG = "ToggleReceiver";
010            @Override
011            public void onReceive(Context context, Intent intent) {
012                    Log.d(TAG, "onReceive " + intent.getAction());
013                    if (MeshTetherApp.ACTION_TOGGLE.equals(intent.getAction())) {
014                            // potential race conditions, but they are benign
015                            MeshService service = MeshService.singleton;
016                            //Log.d(TAG, "service " + ((service == null) ? "null" : "present"));
017                            if (service != null) {
018                                    if (!intent.getBooleanExtra("start", false)) {
019                                            Log.d(TAG, "stop");
020                                            service.stopRequest();
021                                    }
022                            } else {
023                                    if (intent.getBooleanExtra("start", true)) {
024                                            Log.d(TAG, "start");
025                                            context.startService(new Intent(context, MeshService.class));
026                                    }
027                            }
028                    } else if (MeshTetherApp.ACTION_CHECK.equals(intent.getAction())) {
029                            // FIXME: this is the most inefficient way of finding out the state
030                            MeshService service = MeshService.singleton;
031                            int state = (service != null) ? service.getState() : MeshService.STATE_STOPPED;
032                            MeshTetherApp.broadcastState(context, state);
033                    }
034            }
035    }
