019    package net.commotionwireless.meshtether;
020    
021    import android.content.Context;
022    import android.preference.EditTextPreference;
023    import android.text.method.DigitsKeyListener;
024    import android.util.AttributeSet;
025    import android.view.View;
026    import android.widget.EditText;
027    import android.widget.Toast;
028    
029    /**
030     * EditTextPreference that allows IP addresses only
031     */
032    public class MeshIPPreference extends EditTextPreference {
033            public MeshIPPreference(Context context, AttributeSet attrs, int defStyle) { super(context, attrs, defStyle); }
034            public MeshIPPreference(Context context, AttributeSet attrs) { super(context, attrs); }
035            public MeshIPPreference(Context context) { super(context); }
036    
037            @Override
038            protected void onAddEditTextToDialogView(View dialogView, EditText editText) {
039                    editText.setKeyListener(DigitsKeyListener.getInstance("0123456789."));
040                    super.onAddEditTextToDialogView(dialogView, editText);
041            }
042    
043            public static boolean validate(String addr) {
044                    if (addr.matches("^[0-9]{1,3}$"))
045                    {
046                            int prefix = Integer.valueOf(addr).intValue();
047                            if (prefix > 0 && prefix < 255)
048                                    return true;
049                            else
050                                    return false;
051                    }
052                    try {
053                            //if(addr.length() == 0) // don't want empty address
054                            //      return false;
055                            if(java.net.InetAddress.getByName(addr) == null)
056                                    return false;
057                    } catch (java.net.UnknownHostException e) {
058                            return false;
059                    }
060                    return true;
061            }
062    
063            @Override
064            protected void onDialogClosed(boolean positiveResult) {
065                    if (positiveResult) {
066                            // verify now that it's an IP
067                            String addr = getEditText().getText().toString();
068                            if (!validate(addr)) {
069                                    Toast.makeText(getContext(),
070                                                    getContext().getString(R.string.invalidmeship),
071                                                    Toast.LENGTH_SHORT).show();
072                                    positiveResult = false;
073                            }
074                    }
075                    super.onDialogClosed(positiveResult);
076            }
077    }
