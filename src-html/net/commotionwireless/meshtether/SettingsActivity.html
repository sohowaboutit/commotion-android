019    package net.commotionwireless.meshtether;
020    
021    import android.os.Bundle;
022    import android.preference.EditTextPreference;
023    import android.preference.ListPreference;
024    import android.preference.Preference;
025    import android.preference.PreferenceActivity;
026    import android.widget.Toast;
027    
028    public class SettingsActivity extends PreferenceActivity implements Preference.OnPreferenceChangeListener {
029            final static int[] prefids = {
030                    R.string.if_lan, R.string.if_wan,
031                    R.string.adhoc_ip, R.string.adhoc_netmask, R.string.adhoc_dns_server, R.string.lan_essid, R.string.lan_bssid, R.string.lan_channel,
032                    R.string.lan_script
033            };
034            final static int[] checks = { R.string.lan_wext };
035    
036            private void setSummary(Preference p, CharSequence s) {
037                    if ((s != null) && (s.length() > 0)) {
038                            p.setSummary(getString(R.string.current) + s);
039                    } else {
040                            p.setSummary(null);
041                    }
042            }
043    
044            @Override
045            protected void onCreate(Bundle savedInstanceState) {
046                    super.onCreate(savedInstanceState);
047                    addPreferencesFromResource(R.xml.preferences);
048    
049                    for (int i = 0; i < prefids.length; ++i) {
050                            Preference pref = findPreference(getString(prefids[i]));
051                            pref.setOnPreferenceChangeListener(this);
052                            if (ListPreference.class.isInstance(pref)) {
053                                    ListPreference preference = (ListPreference) pref;
054                                    setSummary(preference, preference.getValue());
055                            } else if (EditTextPreference.class.isInstance(pref)) {
056                                    EditTextPreference preference = (EditTextPreference) pref;
057                                    setSummary(preference, preference.getText());
058                            }
059                    }
060                    for (int i = 0; i < checks.length; ++i) {
061                            Preference pref = findPreference(getString(checks[i]));
062                            pref.setOnPreferenceChangeListener(this);
063                    }
064            }
065    
066            @Override
067            public boolean onPreferenceChange(Preference pref, Object newValue) {
068                    String key = pref.getKey();
069                    if (key == null) return true;
070    
071                    if (((MeshTetherApp)getApplication()).isRunning()) {
072                            Toast.makeText(this, getString(R.string.restartneeded), Toast.LENGTH_SHORT).show();
073                    }
074    
075                    if (ListPreference.class.isInstance(pref) || EditTextPreference.class.isInstance(pref)) {
076                            setSummary(pref, (String)newValue);
077                    } // else don't update summary
078                    return true;
079            }
080    }
