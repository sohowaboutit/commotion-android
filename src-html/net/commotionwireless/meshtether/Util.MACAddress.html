026            static class StyledStringBuilder extends android.text.SpannableStringBuilder {
027                    public StyledStringBuilder() { super(); }
028                    private StyledStringBuilder append(Object obj, String s) {
029                            append(s).setSpan(obj, length()-s.length(), length(), 0);
030                            return this;
031                    }
032                    public StyledStringBuilder append(android.text.style.TextAppearanceSpan obj, String s) {
033                            return append((Object)obj, s);
034                    }
035                    public StyledStringBuilder append(int color, String s) {
036                            return append(new android.text.style.ForegroundColorSpan(color), s);
037                    }
038            }

040            public static class MACAddress {
041                    public final static int LENGTH = 6;
042                    public final byte[] value = new byte[LENGTH];
043                    public static MACAddress parse(String s) {
044                            MACAddress addr = new MACAddress();
045                            String[] parts = s.split(":|-|\\.");
046                            if (parts.length != LENGTH)
047                                    return null;
048                            try {
049                                    for (int i = 0; i < LENGTH; ++i) {
050                                            addr.value[i] = (byte)Integer.parseInt(parts[i], 16);
051                                    }
052                            } catch (NumberFormatException e) {
053                                    Log.e(MeshTetherApp.TAG, "Unable to parse "+ s, e);
054                                    return null;
055                            }
056                            return addr;
057                    }
058                    @Override
059                    public String toString() {
060                            byte[] v = value;
061                            return String.format("%02x:%02x:%02x:%02x:%02x:%02x",
062                                            v[0], v[1], v[2], v[3], v[4], v[5]);
063                    }
064                    public int getOctet(int octet) {
065                            return Math.abs(value[octet-1]);
066                    }
067            }

069            public static class TrafficData {
070                    long rx_bytes;
071                    long rx_pkts;
072                    long tx_bytes;
073                    long tx_pkts;
074                    void diff(TrafficData ref, TrafficData cur) {
075                            rx_bytes = cur.rx_bytes - ref.rx_bytes;
076                            rx_pkts  = cur.rx_pkts  - ref.rx_pkts;
077                            tx_bytes = cur.tx_bytes - ref.tx_bytes;
078                            tx_pkts  = cur.tx_pkts  - ref.tx_pkts;
079                    }
080                    void minus(TrafficData ref) {
081                            diff(ref, this);
082                    }
083                    void div(long val) {
084                            if (val == 0) return;
085                            rx_bytes /= val;
086                            rx_pkts /= val;
087                            tx_bytes /= val;
088                            tx_pkts /= val;
089                    }
090            }
091    
092            public static class TrafficStats {
093                    private TrafficData start = new TrafficData();
094                    TrafficData total = new TrafficData();
095                    TrafficData rate  = new TrafficData();
096                    private long t_last = 0;
097    
098                    void init(TrafficData td) {
099                            start = td;
100                            t_last = new java.util.Date().getTime(); // in ms
101                    }
102                    void update(TrafficData td) {
103                            td.minus(start);
104                            rate.diff(total, td);
105                            total = td;
106                            long now = new java.util.Date().getTime();
107                            rate.div((now - t_last) / 1000); // per second
108                            t_last = now;
109                    }
110            }

112            public static TrafficData fetchTrafficData(String device) {
113                    // Returns traffic usage for all interfaces starting with 'device'.
114                    TrafficData d = new TrafficData();
115                    if (device == "")
116                            return d;
117                    for (String line : readLinesFromFile("/proc/net/dev")) {
118                            if (line.startsWith(device)) {
119                                    line = line.replace(':', ' ');
120                                    String[] values = line.split(" +");
121                                    d.rx_bytes += Long.parseLong(values[1]);
122                                    d.rx_pkts  += Long.parseLong(values[2]);
123                                    d.tx_bytes += Long.parseLong(values[9]);
124                                    d.tx_pkts  += Long.parseLong(values[10]);
125                            }
126                    }
127                    return d;
128            }
129    
130            public static ArrayList<String> readLinesFromFile(String filename) {
131                    ArrayList<String> lines = new ArrayList<String>();
132                    try {
133                            java.io.BufferedReader br = toReader(new java.io.FileInputStream(filename));
134                            String line;
135                            while((line = br.readLine()) != null) {
136                                    lines.add(line.trim());
137                            }
138                    } catch (Exception e) {
139                            return null;
140                    }
141                    return lines;
142            }
143    
144            public static java.io.BufferedReader toReader(java.io.InputStream is) {
145                    return new java.io.BufferedReader(new java.io.InputStreamReader(is), 8192);
146            }
