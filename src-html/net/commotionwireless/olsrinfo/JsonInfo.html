051            final Set<String> supportedCommands = new HashSet<String>(
052                            Arrays.asList(new String[] {
053                                            // combined reports
054                                            "all", // all of the JSON info
055                                            "runtime", // all of the runtime status reports
056                                            "startup", // all of the startup config reports
057                                            // individual runtime reports
058                                            "gateways", // gateways
059                                            "hna", // Host and Network Association

061                                            "links", // links

065                                            "topology", // mesh network topology

067                                            // the rest don't change at runtime, so they're separate
068                                            "config", // the current running config info
069                                            "plugins", // loaded plugins and their config
070                                            // only non-JSON output, can't be combined with others
071                                            "olsrd.conf", // current config in olsrd.conf format
072                            }));

086            private boolean isCommandStringValid(String cmdString) {
087                    boolean isValid = true;
088                    if (!cmdString.equals(lastCommand)) {
089                            lastCommand = cmdString;
090                            for (String s : cmdString.split("/")) {
091                                    if ( !s.equals("") && !supportedCommands.contains(s)) {
092                                            System.out.println("Unsupported command: " + s);
093                                            isValid = false;
094                                    }
095                            }
096                    }
097                    return isValid;
098            }

100            /**
101             * Request a reply from the jsoninfo plugin via a network socket.
102             * 
103             * @param The command to query jsoninfo with
104             * @return A String array of the result, line-by-line
105             * @throws IOException when it cannot get a result.
106             */
107            String[] request(String req) throws IOException {
108                    Socket sock = null;
109                    BufferedReader in = null;
110                    PrintWriter out = null;
111                    List<String> retlist = new ArrayList<String>();
112    
113                    try {
114                            sock = new Socket(host, port);
115                            in = new BufferedReader(new InputStreamReader(sock.getInputStream()), 8192);
116                            out = new PrintWriter(sock.getOutputStream(), true);
117                    } catch (UnknownHostException e) {
118                            System.err.println("Unknown host: " + host);
119                            return new String[0];
120                    } catch (IOException e) {
121                            System.err.println("Couldn't get I/O for socket to " + host + ":"
122                                            + Integer.toString(port));
123                            return new String[0];
124                    }
125                    out.println(req);
126                    String line;
127                    while ((line = in.readLine()) != null) {
128                            if (!line.equals(""))
129                                    retlist.add(line);
130                    }
131                    // the jsoninfo plugin drops the connection once it outputs
132                    out.close();
133                    in.close();
134                    sock.close();
135    
136                    return retlist.toArray(new String[retlist.size()]);
137            }
138    
139            /**
140             * Send a command to the jsoninfo plugin and get the raw dump back.
141             * 
142             * @param The command to query jsoninfo with
143             * @return The complete JSON from jsoninfo as single String
144             */
145            public String command(String cmdString) {
146                    String[] data = new String[0];
147                    String ret = "";
148    
149                    isCommandStringValid(cmdString);
150                    try {
151                            data = request(cmdString);
152                    } catch (IOException e) {
153                            System.err.println("Failed to read data from " + host + ":"
154                                            + Integer.toString(port));
155                    }
156                    for (String s : data) {
157                            ret += s + "\n";
158                    }
159                    return ret;
160            }
161    
162            /**
163             * Query the jsoninfo plugin over a network socket and return the results 
164             * parsed into Java objects.
165             * 
166             * @param The command to query jsoninfo with
167             * @return The complete JSON reply parsed into Java objects.
168             */
169            public OlsrDataDump parseCommand(String cmd) {
170                    if (mapper == null)
171                            mapper = new ObjectMapper();
172                    OlsrDataDump ret = new OlsrDataDump();
173                    try {
174                            String dump = command(cmd);
175                            if (! dump.contentEquals(""))
176                                    ret = mapper.readValue(dump, OlsrDataDump.class);
177                            ret.setRaw(dump);
178                    } catch (JsonParseException e) {
179                            // TODO Auto-generated catch block
180                            e.printStackTrace();
181                    } catch (JsonMappingException e) {
182                            // TODO Auto-generated catch block
183                            e.printStackTrace();
184                    } catch (IOException e) {
185                            // TODO Auto-generated catch block
186                            e.printStackTrace();
187                    }
188    
189                    // change nulls to blank objects so you can use this result in a for()
190                    if (ret.config == null)
191                            ret.config = new Config();
192                    if (ret.gateways == null)
193                            ret.gateways = Collections.emptyList();
194                    if (ret.hna == null)
195                            ret.hna = Collections.emptyList();
196                    if (ret.interfaces == null)
197                            ret.interfaces = Collections.emptyList();
198                    if (ret.links == null)
199                            ret.links = Collections.emptyList();
200                    if (ret.mid == null)
201                            ret.mid = Collections.emptyList();
202                    if (ret.neighbors == null)
203                            ret.neighbors = Collections.emptyList();
204                    if (ret.topology == null)
205                            ret.topology = Collections.emptyList();
206                    if (ret.plugins == null)
207                            ret.plugins = Collections.emptyList();
208                    if (ret.routes == null)
209                            ret.routes = Collections.emptyList();
210                    return ret;
211            }

343            /**
344             * for testing from the command line
345             */
346            public static void main(String[] args) throws IOException {
347                    JsonInfo jsoninfo = new JsonInfo();
348                    OlsrDataDump dump = jsoninfo.all();
349                    System.out.println("gateways:");
350                    for (Gateway g : dump.gateways)
351                            System.out.println("\t" + g.ipAddress);
352                    System.out.println("hna:");
353                    for (HNA h : dump.hna)
354                            System.out.println("\t" + h.destination);
355                    System.out.println("Interfaces:");
356                    for (Interface i : dump.interfaces)
357                            System.out.println("\t" + i.name);
358                    System.out.println("Links:");
359                    for (Link l : dump.links)
360                            System.out.println("\t" + l.localIP + " <--> " + l.remoteIP);
361                    System.out.println("MID:");
362                    for (MID m : dump.mid)
363                            System.out.println("\t" + m.ipAddress);
364                    System.out.println("Neighbors:");
365                    for (Neighbor n : dump.neighbors)
366                            System.out.println("\t" + n.ipv4Address);
367                    System.out.println("Plugins:");
368                    for (Plugin p : dump.plugins)
369                            System.out.println("\t" + p.plugin);
370                    System.out.println("Routes:");
371                    for (Route r : dump.routes)
372                            System.out.println("\t" + r.destination);
373                    System.out.println("Topology:");
374                    for (Node node : dump.topology)
375                            System.out.println("\t" + node.destinationIP);
376            }
