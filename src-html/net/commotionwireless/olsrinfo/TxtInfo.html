017    /**
018     * Scrape the output of <tt>olsrd</tt>'s txtinfo plugin into
019     * arrays of Strings.
020     * Written as part of the Commotion Wireless project
021     * 
022     * @author Hans-Christoph Steiner <hans@eds.org>
023     * @see <a href="https://code.commotionwireless.net/projects/commotion/wiki/OLSR_Configuration_and_Management">OLSR Configuration and Management</a>
024     */
025    public class TxtInfo {
026    
027            String host = "127.0.0.1";
028            int port = 2006;
029    
030            public TxtInfo() {
031            }
032    
033            public TxtInfo(String sethost) {
034                    host = sethost;
035            }
036    
037            public TxtInfo(String sethost, int setport) {
038                    host = sethost;
039                    port = setport;
040            }
041    
042            public String[] request(String req) throws IOException {
043                    Socket sock = null;
044                    BufferedReader in = null;
045                    PrintWriter out = null;
046                    List<String> retlist = new ArrayList<String>();
047    
048                    try {
049                            sock = new Socket(host, port);
050                            in = new BufferedReader(new InputStreamReader(sock.getInputStream()), 8192);
051                            out = new PrintWriter(sock.getOutputStream(), true);
052                    } catch (UnknownHostException e) {
053                            throw new IOException();
054                    } catch (IOException e) {
055                            System.err.println("Couldn't get I/O for socket to " + host + ":"
056                                            + Integer.toString(port));
057                    }
058                    out.println(req);
059                    String line;
060                    while ((line = in.readLine()) != null) {
061                            if (!line.equals(""))
062                                    retlist.add(line);
063                    }
064                    // the txtinfo plugin drops the connection once it outputs
065                    out.close();
066                    in.close();
067                    sock.close();
068    
069                    return retlist.toArray(new String[retlist.size()]);
070            }
071    
072            public String[][] command(String cmd) {
073                    String[] data = null;
074    
075                    final Set<String> supportedCommands = new HashSet<String>(
076                                    Arrays.asList(new String[] { "/2ho", // two-hop neighbors
077                                                    "/con", // conf file
078                                                    "/gat", // gateways
079                                                    "/hna", // Host and Network Association
080                                                    "/int", // network interfaces
081                                                    "/lin", // links
082                                                    "/mid", // MID
083                                                    "/nei", // neighbors
084                                                    "/rou", // routes
085                                                    "/top", // topology
086                                    }));
087                    if (!supportedCommands.contains(cmd))
088                            System.out.println("Unsupported command: " + cmd);
089    
090                    try {
091                            data = request(cmd);
092                    } catch (IOException e) {
093                            System.err.println("Couldn't get I/O for socket to " + host + ":"
094                                            + Integer.toString(port));
095                    }
096                    int startpos = -1;
097                    for (int i = 0; i < data.length; i++) {
098                            if (data[i].startsWith("Table: ")) {
099                                    startpos = i + 2;
100                                    break;
101                            }
102                    }
103                    if (startpos >= data.length || startpos == -1)
104                            return new String[0][0];
105                    int fields = data[startpos + 1].split("\t").length;
106                    String[][] ret = new String[data.length - startpos][fields];
107                    for (int i = 0; i < ret.length; i++)
108                            ret[i] = data[i + startpos].split("\t");
109                    return ret;
110            }
111    
112            /**
113             * 2-hop neighbors on the mesh
114             * 
115             * @return array of per-IP arrays of IP address, SYM, MPR, MPRS,
116             *         Willingness, and 2 Hop Neighbors
117             */
118            public String[][] twohop() {
119                    return command("/2ho");
120            }
121    
122            /**
123             * immediate neighbors on the mesh
124             * 
125             * @return array of per-IP arrays of IP address, SYM, MPR, MPRS,
126             *         Willingness, and 2 Hop Neighbors
127             */
128            public String[][] neighbors() {
129                    return command("/nei");
130            }
131    
132            /**
133             * direct connections on the mesh, i.e. nodes with direct IP connectivity
134             * via Ad-hoc
135             * 
136             * @return array of per-IP arrays of Local IP, Remote IP, Hysteresis, LQ,
137             *         NLQ, and Cost
138             */
139            public String[][] links() {
140                    return command("/lin");
141            }
142    
143            /**
144             * IP routes to nodes on the mesh
145             * 
146             * @return array of per-IP arrays of Destination, Gateway IP, Metric, ETX,
147             *         and Interface
148             */
149            public String[][] routes() {
150                    return command("/rou");
151            }
152    
153            /**
154             * Host and Network Association (for supporting dynamic internet gateways)
155             * 
156             * @return array of per-IP arrays of Destination and Gateway
157             */
158            public String[][] hna() {
159                    return command("/hna");
160            }
161    
162            /**
163             * Multiple Interface Declaration
164             * 
165             * @return array of per-IP arrays of IP address and Aliases
166             */
167            public String[][] mid() {
168                    return command("/mid");
169            }
170    
171            /**
172             * topology of the whole mesh
173             * 
174             * @return array of per-IP arrays of Destination IP, Last hop IP, LQ, NLQ,
175             *         and Cost
176             */
177            public String[][] topology() {
178                    return command("/top");
179            }
180    
181            /**
182             * the network interfaces that olsrd is aware of
183             * 
184             * @return array of per-IP arrays of Destination IP, Last hop IP, LQ, NLQ,
185             *         and Cost
186             */
187            public String[][] interfaces() {
188                    return command("/int");
189            }
190    
191            /**
192             * the gateways to other networks that this node knows about
193             * 
194             * @return array of per-IP arrays of Status, Gateway IP, ETX, Hopcount,
195             *         Uplink, Downlink, IPv4, IPv6, Prefix
196             */
197            public String[][] gateways() {
198                    return command("/gat");
199            }

201            /**
202             * for testing from the command line
203             */

205                    TxtInfo txtinfo = new TxtInfo();
206                    System.out.println("NEIGHBORS----------");
207                    for (String[] s : txtinfo.neighbors()) {
208                            for (String t : s)
209                                    System.out.print(t + ",");
210                            System.out.println();
211                    }
212                    System.out.println("LINKS----------");
213                    for (String[] s : txtinfo.links()) {
214                            for (String t : s)
215                                    System.out.print(t + ",");
216                            System.out.println();
217                    }
218                    System.out.println("ROUTES----------");
219                    for (String[] s : txtinfo.routes()) {
220                            for (String t : s)
221                                    System.out.print(t + ",");
222                            System.out.println();
223                    }
224                    System.out.println("HNA----------");
225                    for (String[] s : txtinfo.hna()) {
226                            for (String t : s)
227                                    System.out.print(t + ",");
228                            System.out.println();
229                    }
230                    System.out.println("MID----------");
231                    for (String[] s : txtinfo.mid()) {
232                            for (String t : s)
233                                    System.out.print(t + ",");
234                            System.out.println();
235                    }
236                    System.out.println("TOPOLOGY----------");
237                    for (String[] s : txtinfo.topology()) {
238                            for (String t : s)
239                                    System.out.print(t + ",");
240                            System.out.println();
241                    }
242                    System.out.println("INTERFACES----------");
243                    for (String[] s : txtinfo.interfaces()) {
244                            for (String t : s)
245                                    System.out.print(t + ",");
246                            System.out.println();
247                    }
248                    System.out.println("GATEWAYS----------");
249                    for (String[] s : txtinfo.gateways()) {
250                            for (String t : s)
251                                    System.out.print(t + ",");
252                            System.out.println();
253                    }
254            }
