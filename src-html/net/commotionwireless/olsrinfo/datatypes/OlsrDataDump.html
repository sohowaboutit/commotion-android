001    package net.commotionwireless.olsrinfo.datatypes;
002    
003    import java.util.Collection;
004    
005    /**
006     * A class representing the complete dump of information from <tt>olsrd</tt>'s
007     * jsoninfo plugin.
008     * 
009     * Written as part of the Commotion Wireless project
010     * 
011     * @author Hans-Christoph Steiner <hans@eds.org>
012     * @see <a href="https://code.commotionwireless.net/projects/commotion/wiki/OLSR_Configuration_and_Management">OLSR Configuration and Management</a>
013     */
014    public class OlsrDataDump {
015            public Config config;
016            public Collection<Gateway> gateways;
017            public Collection<HNA> hna;
018            public Collection<Interface> interfaces;
019            public Collection<Link> links;
020            public Collection<MID> mid;
021            public Collection<Neighbor> neighbors;
022            public Collection<Node> topology;
023            public Collection<Plugin> plugins;
024            public Collection<Route> routes;
025            public int systemTime = 0;
026            public int timeSinceStartup = 0;
027            public String uuid = "";
028    
029            String raw = "";
030    
031            public void setRaw(String s) {
032                    raw = s;
033            }
034            
035            public String toString(){
036                    return raw;
037            }
038    }
